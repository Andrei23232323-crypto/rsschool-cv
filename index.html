String.prototype.toAlternatingCase = function () {

let nowStr = '';
let arr = this.split('');

  for (let sim of arr) {
    if (sim === sim.toUpperCase()) {
      nowStr += sim.toLowerCase();
    } else {
      nowStr += sim.toUpperCase();
    }
  }
  return (nowStr);
}
